#include <stdio.h>
int main(){
int num;

do
{
    int sum=0;
    printf("Enter an even integer number: ");
    scanf("%d", &num);
    if ( num % 2 != 0)
    {
        printf("It's not an even integer");
    }
   for (int i = 0; i <= num; i++)
   {
    if (i % 2 == 0)
    {
         sum= sum + i;
    }
    
   }
   
    printf("Sum of all even values from 0 to %d is: %d\n", num, sum );

}
 while (num % 2 == 0);

    
    return 0;
}
