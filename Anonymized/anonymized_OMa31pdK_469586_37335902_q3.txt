#include <stdio.h>

int main () {
    
    int num;

    do {
        printf("Enter a number (1 - 10): ");
        scanf("%d", &num);
    } while (num < 1 || num > 10);


    for (int i = 1; i <= num; i++) {

        for (int j = 1; j < num; j++) {
            if () {
                // Print out block of white space
                for (int j = 1; j < num; j++) {
                    printf(" ");
                }

            }

            else {
                // Print out the RHS first
                 for (int j = 1; j < num; j++) {
                    printf("%d", j);
                }

            }
        }

        printf("10"); // Middle column of 10


        // Print out LHS 
        for (int j = num - 1; j >= 1; j--) {
            printf("%d", j);
        }
    
        printf("\n"); 
    }


      return 0;
}
