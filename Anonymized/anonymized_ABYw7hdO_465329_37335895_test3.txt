#include <stdio.h>
int main(){
    int num;
    printf("Enter an integer between 1 and 10: ");
    scanf("%d",&num);
    for(int row =1;row <=num;row++){
        for( int block =1;block <=num-row;block++){
            printf("    ");
        }
        for (int val = num- row+1;val<=num;val++){
            printf(" %2d ",val);
        }
        
        printf("\n");
    }

    return 0;
}
