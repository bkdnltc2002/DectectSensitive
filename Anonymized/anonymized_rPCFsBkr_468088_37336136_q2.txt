#include <stdio.h>
int main (){
    int num;
    int sum =0;
    int i=0;
    while (1){
        printf ("Enter an even integer: ");
        scanf ("%d", &num);
        if (num % 2 != 0){ 
            printf("It is not an even integer.");
            break;            
        }
        for (i = 0 ; i <= num ; i +=2){
            sum += i;
        }
        printf ("Sum of all even values from 0 to %d is: %d\n", num , sum);

    }
    
    return 0;
}
