        }
while ( num <=1 || num >10);


for (int row = 1; row <= num; row++){
    // block of space
    for(int space = 1; space <= num - row; space ++){

    }
// first half of the pyramid

    for(int col= num + 1 - row; col <= num; col ++){

    }

    // second half of the pyramid
     for (int col = num -1; col > num -row ; col --){
        printf("%4d", col);
    }

}
