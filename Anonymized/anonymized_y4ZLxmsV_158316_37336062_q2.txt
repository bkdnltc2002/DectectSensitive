#include <stdio.h>

int main(void) {
    int num;
    
    int sum = 0;

    do {
        printf("Enter an even integer: ");
        scanf("%d", &num);

        if (num % 2 != 0) {
            printf("It is not an even integer.");
            break;
        }
        
        for (int even_num = 0; even_num <= num; even_num++) {
            sum = even_num + even_num;
        }

        printf("Sum of all even values from 0 to %d is: %d", num, sum);
        break;
    } while (num % 2 == 0);
    
    return 0;
}
