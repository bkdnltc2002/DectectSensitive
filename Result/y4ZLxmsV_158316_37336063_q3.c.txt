#include <stdio.h>

int main(void) {
    int num;
    printf("Enter a number (1 - 10): ");
    scanf("%d", &num);

    for (int col = 1; col <= num; col++) {
        if (col == 1) {
            
            // White spaces
            for (int block = 1; block <= num - col - 1; block++) {
                printf("    ");
            }
            
        }
    }

    return 0;
}
